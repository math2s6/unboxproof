$ch = curl_init('https://your-app.onrender.com/api/b2b/orders');
curl_setopt_array($ch, [
  CURLOPT_POST => true,
  CURLOPT_HTTPHEADER => [
    'Content-Type: application/json',
    'X-API-Key: ' . $_ENV['UNBOXPROOF_API_KEY']
  ],
  CURLOPT_POSTFIELDS => json_encode([
    'external_order_id' => 'CMD-12345',
    'customer_email'   => 'client@example.com',
    'product_name'     => 'Sneakers Pro X',
    'order_amount'     => 149.00
  ]),
  CURLOPT_RETURNTRANSFER => true
]);
$order = json_decode(curl_exec($ch));
